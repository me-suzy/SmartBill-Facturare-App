Îți înțeleg preocuparea. Problema este că cifrele din tabelul "Lista Clienți" din `clienti.html` sunt prea apropiate de butoanele de editare (`✏️`) și ștergere (`🗑️`), ceea ce poate afecta usability-ul. Pentru a rezolva acest lucru, putem adăuga spațiere suplimentară între coloana "Sold (RON)" și coloana "Acțiuni" prin ajustarea stilurilor CSS.

### Soluție
Trebuie să modificăm CSS-ul pentru a adăuga padding sau margini în tabel, în special între coloana "Sold (RON)" și butoane. De asemenea, putem ajusta dimensiunea și poziționarea butoanelor pentru a le face mai distincte.

#### Pași
1. **Adaugă spațiere în tabel**: Folosește `padding-right` în celulele din coloana "Sold (RON)" și ajustează lățimea coloanei "Acțiuni".
2. **Stilizează butoanele**: Adaugă margini sau padding la butoane pentru a le separa mai bine de text.
3. **Actualizează `clienti.html` și `style.css` în consecință**.

#### 3. Actualizează `clienti.js`
Modifică JavaScript-ul pentru a adăuga atributul `data-stat="sold"` la generarea rândurilor tabelului.

```javascript
document.addEventListener('DOMContentLoaded', function() {
    const clientiTable = document.querySelector('#clientiTable tbody');
    const totalClienti = document.querySelector('[data-stat="total_clienti"] .stat-value');
    const clientiActivi = document.querySelector('[data-stat="clienti_activi"] .stat-value');
    const soldTotal = document.querySelector('[data-stat="sold_total"] .stat-value');
    const clientiDatorii = document.querySelector('[data-stat="clienti_datorii"] .stat-value');
    const clientModal = document.getElementById('clientModal');
    const clientForm = document.getElementById('clientForm');
    const modalTitle = document.getElementById('modalTitle');

    // Load clients
    function loadClienti() {
        fetch('/api/get_clienti')
            .then(response => response.json())
            .then(clienti => {
                clientiTable.innerHTML = '';
                clienti.forEach(client => {
                    const row = document.createElement('tr');
                    row.innerHTML = `
                        <td>${client.ID}</td>
                        <td>${client.CIF || ''}</td>
                        <td>${client.Nume || ''}</td>
                        <td>${client.Email || ''}</td>
                        <td>${client.Telefon || ''}</td>
                        <td>${client.Localitate || ''}</td>
                        <td data-stat="sold">${(client.Sold_Curent || 0).toFixed(2)}</td>
                        <td>
                            <button class="action-btn edit-btn" onclick="editClient(${client.ID})">✏️</button>
                            <button class="action-btn delete-btn" onclick="deleteClient(${client.ID})">🗑️</button>
                        </td>
                    `;
                    clientiTable.appendChild(row);
                });

                // Update statistics
                totalClienti.textContent = clienti.length;
                clientiActivi.textContent = clienti.filter(c => c.Status === 'Activ').length;
                soldTotal.textContent = clienti.reduce((sum, c) => sum + (c.Sold_Curent || 0), 0).toFixed(2);
                clientiDatorii.textContent = clienti.filter(c => (c.Sold_Curent || 0) > 0).length;
            })
            .catch(error => {
                console.error('Eroare la încărcarea clienților:', error);
                alert('Nu s-au putut încărca clienții: ' + error.message);
            });
    }

    // Open add client modal
    window.openAddClientModal = function() {
        modalTitle.textContent = 'Adaugă Client Nou';
        clientForm.reset();
        clientForm.dataset.clientId = '';
        clientModal.style.display = 'flex';
    };

    // Open edit client modal
    window.editClient = function(clientId) {
        fetch(`/api/get_clienti`)
            .then(response => response.json())
            .then(clienti => {
                const client = clienti.find(c => c.ID === clientId);
                if (client) {
                    modalTitle.textContent = 'Editează Client';
                    document.getElementById('clientCIF').value = client.CIF || '';
                    document.getElementById('clientName').value = client.Nume || '';
                    document.getElementById('clientEmail').value = client.Email || '';
                    document.getElementById('clientPhone').value = client.Telefon || '';
                    document.getElementById('clientCity').value = client.Localitate || '';
                    document.getElementById('clientCounty').value = client.Judet || '';
                    document.getElementById('clientAddress').value = client.Adresa || '';
                    document.getElementById('clientPostalCode').value = client.Cod_Postal || '';
                    document.getElementById('clientWebsite').value = client.Website || '';
                    document.getElementById('clientContactPerson').value = client.Persoana_Contact || '';
                    document.getElementById('clientNotes').value = client.Observatii || '';
                    clientForm.dataset.clientId = clientId;
                    clientModal.style.display = 'flex';
                }
            })
            .catch(error => {
                console.error('Eroare:', error);
                alert('Eroare la încărcarea datelor clientului!');
            });
    };

    // Close modal
    window.closeModal = function() {
        clientModal.style.display = 'none';
        clientForm.reset();
    };

    // Handle form submission
    clientForm.addEventListener('submit', function(e) {
        e.preventDefault();
        const clientId = clientForm.dataset.clientId;
        const clientData = {
            cif: document.getElementById('clientCIF').value,
            nume: document.getElementById('clientName').value,
            email: document.getElementById('clientEmail').value,
            telefon: document.getElementById('clientPhone').value,
            localitate: document.getElementById('clientCity').value,
            judet: document.getElementById('clientCounty').value,
            adresa: document.getElementById('clientAddress').value,
            cod_postal: document.getElementById('clientPostalCode').value,
            website: document.getElementById('clientWebsite').value,
            persoana_contact: document.getElementById('clientContactPerson').value,
            observatii: document.getElementById('clientNotes').value
        };

        const url = clientId ? `/api/update_client/${clientId}` : '/api/add_client';
        const method = clientId ? 'PUT' : 'POST';

        fetch(url, {
            method: method,
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(clientData)
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                alert(clientId ? 'Client actualizat cu succes!' : 'Client adăugat cu succes!');
                loadClienti();
                closeModal();
            } else {
                alert('Eroare: ' + data.message);
            }
        })
        .catch(error => {
            console.error('Eroare:', error);
            alert('Eroare la salvarea clientului!');
        });
    });

    // Delete client
    window.deleteClient = function(clientId) {
        if (confirm('Sigur vrei să ștergi acest client?')) {
            fetch(`/api/delete_client/${clientId}`, {
                method: 'DELETE'
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    alert('Client șters cu succes!');
                    loadClienti();
                } else {
                    alert('Eroare la ștergerea clientului: ' + data.message);
                }
            })
            .catch(error => {
                console.error('Eroare:', error);
                alert('Eroare la ștergerea clientului!');
            });
        }
    };

    // Load clients on page load
    loadClienti();
});
```

**Modificări**:
- Adăugat `data-stat="sold"` la `<td>` pentru coloana "Sold (RON)" în generarea rândurilor tabelului.
